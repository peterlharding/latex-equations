\section{Maxwell's Equations}

\begin{equation*}
	\begin{alignedat}{4}
\div E    & = 0   & \qquad  &  \curl E   & = - \frac{1}{c} \frac{\partial{H}}{{\partial{t}}}   \\[0.5ex]
\div H    & = 0   & \qquad  &  \curl H   & =   \frac{1}{c} \frac{\partial{E}}{{\partial{t}}}              
	\end{alignedat}
\end{equation*}
